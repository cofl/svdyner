rules:
 - plurals: ër
 - possessive: n if vowel, e if consonant

punctuation:
 - ' and - are used ONLY for word concatenation
 - [sentence].
 - [sentence/fragment]… (optional, as in English)
 - [sentence/fragment]— (optional, as in English)
 - ¿[question]?
 - ¡[exclamation]!
 - «» around concept/name
 - ° infix «», rare, usually only for numbers
 - "" around dialogue
 - , to separate connected phrases
 - ; to separate connected groups of phrases
 - / shorthand "or" operator, i.e. "qæra ræ'rá qæerí" -> "qæra/qæerí"
